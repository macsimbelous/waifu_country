on_actions = {
	#ROOT is subject FROM is overlord
	on_puppet = {
		effect = {
			if = {
				limit = {
					FROM = {
						original_tag = HEN
					}
				}
				#add_ideas = wfc_acw_puppet
				if = {
					limit = {
						ROOT = {
							NOT = {
								OR = {
									original_tag = ACW
									original_tag = HEN
									original_tag = MAH
									original_tag = TRP
									original_tag = MMS
									original_tag = MEH
								}
							}
						}
					}
					hen_change_country_leader = yes
				}
			}
		}
	}
	# ROOT is capitulated country, FROM is winner
	on_capitulation = {
		effect = {
			if = {
				limit = {
					tag = JAP
					HEN = {
						has_war_with = JAP
					}
				}
				HEN = {
					set_country_flag = hen_world_conquest_enabled_flag
					remove_ideas = hen_attack_jap_bonus
				}
			}
		}
	}
}
