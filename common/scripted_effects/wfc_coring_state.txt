wfc_coring_state = {
	for_each_loop = {
		array = core_countries
		value = temp_tag
		remove_core_of = temp_var:temp_tag
	}
	add_core_of = ROOT
}

wfl_up_resources_state = {
	set_temp_variable = {
		mult_res = 1
	}
	if = {
		limit = {
			has_state_flag = wfl_up_resources_1
		}
		add_to_temp_variable = {
			mult_res = 1
		}
	}
	if = {
		limit = {
			has_state_flag = wfl_up_resources_2
		}
		add_to_temp_variable = {
			mult_res = 1
		}
	}
	if = {
		limit = {
			has_state_flag = wfl_up_resources_3
		}
		add_to_temp_variable = {
			mult_res = 1
		}
	}
	set_temp_variable = {
		base_res = 8
	}
	multiply_temp_variable = {
		base_res = mult_res
	}
	if = {
		limit = {
			steel > 0
		}
		add_resource = {
			type = steel
			amount = var:base_res
		}
	}
	if = {
		limit = {
			oil > 0
		}
		add_resource = {
			type = oil
			amount = var:base_res
		}
	}
	if = {
		limit = {
			aluminium > 0
		}
		add_resource = {
			type = aluminium
			amount = var:base_res
		}
	}
	if = {
		limit = {
			tungsten > 0
		}
		add_resource = {
			type = tungsten
			amount = var:base_res
		}
	}
	if = {
		limit = {
			chromium > 0
		}
		add_resource = {
			type = chromium
			amount = var:base_res
		}
	}
}

wfl_up_resources_state2 = {

		limit = {
			steel > 0
		}
		
		set_temp_variable = {
			temp_inf_multipler = infrastructure_level
		}
		divide_temp_variable = {
			temp_inf_multipler = 10
		}
		add_to_temp_variable = {
			temp_inf_multipler = 1
		}
		set_temp_variable = {
			temp_steel = resource@steel
		}
		divide_temp_variable = {
			temp_steel = temp_inf_multipler
		}
		multiply_temp_variable = {
			temp_steel = 1.2
		}
		round_temp_variable = temp_steel

			type = steel
