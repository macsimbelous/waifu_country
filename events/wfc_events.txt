	id = wfc_event.1
	title = wfc_event.1.t
	desc = wfc_event.1.d
	picture = GFX_report_event_iww_demonstration
	#is_triggered_only = yes
	fire_only_once = no
	trigger = {
		#always = no
		OR = {
			original_tag = ACW

			original_tag = MAH
			original_tag = TRP
			original_tag = MMS
			original_tag = MEH
		}
		communism > 0.3
		NOT = {
			has_government = communism
		}
	}
	mean_time_to_happen = {
		days = 180
		modifier = {
			factor = 0.6
			communism > 0.4
		}
		modifier = {
			factor = 0.5
			communism > 0.5
		}
	}
	option = {
		name = wfc_event.1.a
		ai_chance = {
			factor = 10
		}
		add_popularity = {
			ideology = communism
			popularity = 0.05
		}
	}
	option = {
		name = wfc_event.1.b
		ai_chance = {
			base = 0
			modifier = {
				has_government = fascism
				add = 100
			}
		}
		add_stability = -0.025
		if = {
			limit = {
				has_government = fascism
			}
			add_popularity = {
				ideology = fascism
				popularity = -0.025
			}
		}
		else_if = {
			limit = {
				has_government = democratic
			}
			add_popularity = {
				ideology = democratic
				popularity = -0.025
			}
		}
		else = {
			add_popularity = {
				ideology = neutrality
				popularity = -0.025
			}
		}
	}
	option = {
		name = wfc_event.1.c
		ai_chance = {
			base = 0
			modifier = {
				has_government = democratic
				add = 100
			}
		}
	}
}
