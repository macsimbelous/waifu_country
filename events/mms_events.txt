	}
}

country_event = {
	id = mms_event.4
	title = mms_event.4.t
	desc = mms_event.4.d
	#picture = GFX_mms_event_01
	fire_only_once = yes
	trigger = {
		original_tag = MMS
		has_power_balance = {
			id = mms_balance_of_power
		}
		power_balance_value = {
			id = mms_balance_of_power
			value > 0.99
		}
	}
	option = {
		name = mms_event.4.a
		ai_chance = {
			base = 75
		}
		kill_country_leader = yes
		add_country_leader_role = {
			character = MMS_02
			promote_leader = yes
			country_leader = {
				ideology = despotism
				expire = "1999.1.1.1"
				traits = {
					wfc_monster
					wfc_demon_lord
					wfc_tyan_militarist
					wfc_patron_of_intelligence
				}
			}
		}
		MMS_02 = {
			promote_leader = yes
		}
		remove_power_balance = {
			id = mms_balance_of_power
		}
		add_war_support = 0.15
		add_political_power = -50
	}
}
